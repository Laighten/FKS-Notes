**思路3：双向指针**

头尾向中间遍历，最后写了如下代码：

```js
//两数之和
var twoSum = function(array,n){
    let i = 0;
    let j = array.length-1;
    const res  = [];
    
    while(i!==j){
        let num1 = array[i];
        let num2 = array[j];
        if(num1+num2 === n){
            res.push(num1);
            res.push(num2);
            return res;
        }else if(num1+num2< n){
            //左侧指针右移
            i +=1;
        }else{
            j-=1;
        }
    }
    
    return res;
}

var testArray = [1,2,3,4,10,30];
var n = 13;
console.log(twoSum(testArray,n))//[3,10]
```
